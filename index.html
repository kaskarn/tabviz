tabviz

Interactive Tables with Embedded Visualizations for R

tabviz turns a data frame into a publication-quality, fully interactive htmlwidget — sortable, filterable, draggable, editable — with SVG / PNG / PDF export that captures exactly what you see on screen.

Hero example

TipTry the interactivity
  • Sort — click any column header.
  • Filter — click the funnel icon and pick a text / numeric / categorical filter.
  • Reorder rows — hover a row, grab the drag handle, drop within the same group.
  • Reorder columns — drag column headers; columns stay within their column group.
  • Edit values — double-click a cell (or a column-group header) to edit inline.
  • Export — the download button writes SVG / PNG matching your current view exactly.

Selling points

Declarative columns. The columns = list(...) API mirrors what you see — column types, groups, and visualizations compose with no hidden state.

WYSIWYG export. The same SVG engine drives both the browser and headless export, so PDFs match the screen pixel-for-pixel after sort / filter / edit.

Built for science. Forest plots, multi-arm effects, p-values, intervals, a curated theme roster (the clinical nejm default plus editorial, design, and expressive presets), and multi-page PDF pagination for long meta-analyses — all first-class.

Fully interactive. Sorting, filtering, drag-to-reorder, inline editing, and Shiny proxy updates ship in the box.

Build any table in a few lines

library(tabviz)

tabviz(data,
  label = "study",
  columns = list(
    col_group("Demographics",
      col_text("site"),
      col_n("n")
    ),
    viz_forest(point = "hr", lower = "lo", upper = "hi",
               scale = "log", null_value = 1),
    col_group("Results",
      col_interval("hr", "lo", "hi", header = "HR (95% CI)"),
      col_pvalue("pvalue", "P")
    )
  ),
  interaction = web_interaction_full()
)

## Hero example

```{r}
#| echo: false
data(glp1_trials)

tabviz(
  glp1_trials,
  label = "study",
  group = "group",
  columns = list(
    col_group("Study Info",
      col_text("drug", "Drug"),
      col_n("n"),
      col_events("events", "n", "Events")
    ),
    viz_forest(
      point = "hr", lower = "lower", upper = "upper",
      scale = "log", null_value = 1,
      axis_label = "Hazard Ratio (95% CI)",
      axis_range = c(0.4, 1.5),
      axis_ticks = c(0.5, 0.75, 1.0, 1.25),
      axis_gridlines = TRUE,
      annotations = list(
        refline(0.85, label = "Pooled HR", style = "dashed", color = "#00407a")
      )
    ),
    col_group("Results",
      col_interval("hr", "lower", "upper", header = "HR (95% CI)"),
      col_pvalue("pvalue", "P", stars = TRUE)
    )
  ),
  row_type = "row_type", row_bold = "row_bold",
  interaction = web_interaction_full(),
  theme = web_theme_nejm(),
  title = "GLP-1 Agonist Cardiovascular Outcomes",
  subtitle = "Major adverse cardiovascular events (MACE)"
)
```

## Feature highlights

### Row groups + column groups together

```{r}
#| echo: false
data(effect_sizes)

tabviz(effect_sizes,
  label = "study",
  group = "treatment",
  columns = list(
    col_group("Trial",
      col_text("region", "Region"),
      col_text("phase", "Phase"),
      col_n("n")
    ),
    viz_forest(point = "hr", lower = "lower", upper = "upper",
               scale = "log", null_value = 1,
               axis_label = "Hazard Ratio"),
    col_group("Effect",
      col_interval("hr", "lower", "upper", header = "HR (95% CI)"),
      col_badge("direction", "Direction",
        variants = list(
          `Favors Treatment` = "success",
          `Favors Control` = "error",
          `No Effect` = "muted"
        ))
    )
  ),
  interaction = web_interaction_full(),
  theme = web_theme_nejm(),
  title = "Row Groups x Column Groups",
  subtitle = "Click a group header to collapse; drag within any group to reorder"
)
```

### Multiple effects per row

```{r}
#| echo: false
dose <- data.frame(
  endpoint = c("Overall Survival", "Progression-Free Survival",
               "Objective Response", "Quality of Life"),
  low_hr  = c(0.88, 0.82, 0.85, 0.92),
  low_lo  = c(0.74, 0.68, 0.70, 0.78),
  low_hi  = c(1.05, 0.99, 1.03, 1.08),
  mid_hr  = c(0.76, 0.71, 0.74, 0.83),
  mid_lo  = c(0.62, 0.58, 0.60, 0.69),
  mid_hi  = c(0.93, 0.87, 0.91, 1.00),
  high_hr = c(0.65, 0.60, 0.62, 0.75),
  high_lo = c(0.51, 0.46, 0.48, 0.61),
  high_hi = c(0.83, 0.78, 0.80, 0.92)
)

tabviz(dose, label = "endpoint",
  columns = list(
    viz_forest(
      effects = list(
        effect_forest("low_hr", "low_lo", "low_hi", label = "Low"),
        effect_forest("mid_hr", "mid_lo", "mid_hi", label = "Standard"),
        effect_forest("high_hr", "high_lo", "high_hi", label = "High")
      ),
      scale = "log", null_value = 1,
      axis_label = "Hazard Ratio"
    )
  ),
  interaction = web_interaction(enable_sort = TRUE, enable_filters = TRUE,
                                enable_reorder_rows = TRUE),
  theme = web_theme_nejm(),
  title = "Dose-Response (three arms per row)"
)
```

### Rich dashboards — heatmaps, progress, sparklines, badges

```{r}
#| echo: false
data(airline_delays)
april <- airline_delays |>
  filter(month == "Apr") |>
  select(carrier, carrier_type, on_time_pct, satisfaction, flights, delay_vs_avg, trend)

tabviz(april, label = "carrier",
  columns = list(
    col_badge("carrier_type", "Type",
      variants = list(Legacy = "info", `Low-Cost` = "success",
                      `Ultra Low-Cost` = "warning")),
    col_n("flights", "Flights", abbreviate = TRUE),
    col_heatmap("on_time_pct", "On-Time",
      palette = c("#fee2e2", "#dcfce7"),
      min_value = 0.65, max_value = 0.95, decimals = 2, show_value = TRUE),
    col_progress("satisfaction", "Satisfaction", max_value = 5),
    col_numeric("delay_vs_avg", "Delay vs Avg", decimals = 1),
    col_sparkline("trend", "12-Month Trend")
  ),
  interaction = web_interaction_full(),
  theme = web_theme_nejm(),
  title = "Airline Performance (April)"
)
```
